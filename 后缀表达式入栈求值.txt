#include<stdio.h>
#include<stdlib.h>
#include<string.h>
int x1,x2,x3;
typedef struct node
{
    long a[100];
    int top;
}node;
node *p;
void pushstack(int a)
{
    (p->top)++;
    p->a[p->top] = a;
}
int stacktop()
{

    return p->a[p->top];
}
void popdate()
{
    (p->top)--;
}
void pushdate(char *a,int len)
{
    int i = 0;
    char ch;
    for(i;i<len;i++)
    {
        ch = a[i];
        switch(ch)
        {
            case '+':
            {
                x1 = stacktop();
                popdate();
                x2 = stacktop();
                popdate();
                x3 = x1+x2;
                pushstack(x3);
                break;
            }
            case '-':
            {
                x1 = stacktop();
                popdate();
                x2 = stacktop();
                popdate();
                x3 = x1-x2;
                pushstack(x3);
                break;
            }
            case '*':
            {
                x1 = stacktop();
                popdate();
                x2 = stacktop();
                popdate();
                x3 = x1*x2;
                pushstack(x3);
                break;
            }
            case '/':
            {
                x1 = stacktop();
                popdate();
                x2 = stacktop();
                popdate();
                x3 = x1/x2;
                pushstack(x3);
                break;
            }
            default:
            {
                pushstack(ch-'0');
                break;
            }
        }
    }
}
void dateinite()
{
    p->top = -1;
}

int main()
{
    char a[100];
    p = (struct node *)malloc(sizeof(struct node));
    dateinite();
    gets(a);
    pushdate(a,strlen(a));
    printf("%d",p->a[0]);
    free(p);
    return 0;
}
